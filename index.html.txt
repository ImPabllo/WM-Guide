    <!-- JavaScript Logik zur Verbindung mit dem Backend -->
    <script>
        async function ladeAnalyse(matchId) {
            const bereich = document.getElementById('analyse-bereich');
            const textfeld = document.getElementById('analyse-text');
            
            // Lade-Zustand anzeigen
            bereich.classList.remove('hidden');
            textfeld.innerHTML = "⏳ Die KI analysiert gerade die aktuellen Kader, Quoten und News... Bitte warten...";

            try {
                // Anfrage an unser Python-Backend (läuft lokal auf Port 8000)
                const response = await fetch(`https://ss-wm-tipps.onrender.com/api/analysis/${matchId}`);
                const data = await response.json();

                // HTML mit den echten Daten befüllen
                document.getElementById('heim-name').innerText = data.match_info.home;
                document.getElementById('gast-name').innerText = data.match_info.away;
                
                document.getElementById('quote-1').innerText = data.quoten.bet365["1"];
                document.getElementById('quote-x').innerText = data.quoten.bet365["X"];
                document.getElementById('quote-2').innerText = data.quoten.bet365["2"];

                document.getElementById('kader-home').innerText = data.kader.home.join(', ');
                document.getElementById('kader-away').innerText = data.kader.away.join(', ');

                // KI-Text einfügen
                textfeld.innerText = data.analyse;

            } catch (error) {
                textfeld.innerHTML = "❌ Fehler beim Laden der Analyse. Läuft dein Python-Backend?";
                console.error(error);
            }
        }
    </script>
